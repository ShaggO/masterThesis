\begin{frame}{Proposed descriptor}
\begin{align*}
H_j(f_i) = \int F(\x) A_j (\x) P (\x) B(\x; f_i,f) \,\text{d} \x
\end{align*}
where
\begin{itemize}
\item[] $i$ is the bin number
\item[] $j$ is the cell number
\end{itemize}
\end{frame}
%
\begin{frame}{Histogram domain}

\end{align*}
Bin value function $f$

Magnitude function $F$
\end{frame}
